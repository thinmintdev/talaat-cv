```typescript
// Automatic portfolio synchronization
const syncPortfolio = async (username: string) => {
  const [repos, profile, contributions] = await Promise.all([
    github.repos.listForUser({ username }),
    github.users.getByUsername({ username }),
    github.activity.listContributions({ username })
  ]);
  
  return {
    projects: await analyzeProjects(repos.data),
    skills: extractSkills(repos.data),
    activity: processContributions(contributions.data),
    bio: enhanceProfile(profile.data)
  };
};
```
